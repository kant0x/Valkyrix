function tileToWorld(col: number, row: number, tileWidth: number, tileHeight: number): { wx: number; wy: number } {
  return {
    wx: (col - row) * (tileWidth / 2) + tileWidth / 2 + 1152,
    wy: (col + row) * (tileHeight / 2) + tileHeight / 2,
  };
}
