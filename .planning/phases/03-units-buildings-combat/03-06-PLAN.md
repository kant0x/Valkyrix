```typescript
// Call order in update(dt):

buildingSystem.update(dt, gameState);
projectileSystem.update(dt, gameState);
combatSystem.update(dt, gameState);

// Call in render(ctx):
// ... (existing tile map render calls first)
renderer.render(ctx, gameState, cameraCenter, zoom);

// HUD wiring (every frame in update or render):
hud.update({ wave: gameState.waveNumber, health: gameState.citadelHp, resources: gameState.resources, citadelMaxHp: gameState.citadelMaxHp });

// Win/loss (check once per frame after CombatSystem):
if (gameState.phase === 'won' && !winLossShown) { hud.showWinLossOverlay('won'); winLossShown = true; }
if (gameState.phase === 'lost' && !winLossShown) { hud.showWinLossOverlay('lost'); winLossShown = true; }

// Canvas click for tower placement:
canvas.addEventListener('click', (e) => {
  if (!selectedTowerType) return;
  const rect = canvas.getBoundingClientRect();
  const { col, row } = canvasClickToTile(e.clientX - rect.left, e.clientY - rect.top, canvas, cameraCenter, zoom, mapTileWidth, mapTileHeight);
  buildingSystem.placeBuilding(selectedTowerType, col, row, zoneLayer, mapWidth, gameState);
});
```
