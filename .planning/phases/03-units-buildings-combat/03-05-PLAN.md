function worldToScreen(wx: number, wy: number, cameraCenter: {x:number,y:number}, zoom: number): {sx:number, sy:number} {
  return {
    sx: 1280 / 2 + (wx - cameraCenter.x) * zoom,   // GAME_W = 1280
    sy: 506 / 2 + (wy - cameraCenter.y) * zoom,    // GAME_VIEW_H = 506
  };
}
