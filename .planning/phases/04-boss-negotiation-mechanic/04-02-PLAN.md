```typescript
import { GameState } from './game.types';
import { UNIT_DEFS } from './game.types';
import { NegotiationOverlay } from '../screens/NegotiationOverlay';

const BOSS_TRIGGER_SECONDS = 300;
const NEGOTIATION_RESOURCE_REWARD = 120;
const NEGOTIATION_HP_REWARD = 400;
const NEGOTIATION_WAVE_TIMER_FLOOR = 20;
const HORDE_COMPOSITION: Record<string, number> = {
  'heavy-enemy': 8,
  'ranged-enemy': 6,
};

export class BossSystem {
  private overlay: NegotiationOverlay | null = null;
  private negotiationActive = false;

  update(dt: number, state: GameState, container: HTMLElement | null): void { ... }
  handleSuccess(state: GameState): void { ... }
  handleFailure(state: GameState): void { ... }
  forceReset(state: GameState): void { ... }
  private cleanup(): void { ... }
  private enqueueHorde(state: GameState): void { ... }
}
```

In update():
- state.elapsed = (state.elapsed ?? 0) + dt;
- If elapsed < BOSS_TRIGGER_SECONDS → return
- If state.bossNegotiation?.triggered → return (prevents re-fire)
- Build boss Unit object (spread UNIT_DEFS['boss-enemy'], assign fresh id)
- Push to state.units
- Set state.phase = 'negotiation'
- Set state.bossNegotiation = { active: true, triggered: true, scale: 0, attemptsLeft: 3 } synchronously
- If container: create overlay, call mount with onSuccess/onFailure callbacks and initialScale/initialAttempts from bossNegotiation

In enqueueHorde():
- Flatten HORDE_COMPOSITION into SpawnQueueEntry array
- delay = 1.0 + index * 0.4 for each entry
- Append to state.spawnQueue

In handleFailure():
- Enrage the boss unit in state.units (find by role==='boss' && faction==='enemy', multiply def.damage by 1.5)
- Call this.enqueueHorde(state)
- Set state.bossNegotiation = { active: false, triggered: true, outcome: 'failure' }
- Set state.phase = 'playing'
- Call this.cleanup()

In forceReset():
- this.cleanup()
- if (state.bossNegotiation) state.bossNegotiation = { active: false, triggered: false }
- state.elapsed = 0

Do NOT add redundant phase checks beyond what already exists in main.ts.
Follow anti-patterns from RESEARCH.md: always spread UNIT_DEFS, never mutate constants.
