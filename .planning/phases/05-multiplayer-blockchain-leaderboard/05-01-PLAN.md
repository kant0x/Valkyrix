```typescript
/** Payload sent with each kill transaction to Solana devnet via MagicBlock ER */
export interface KillPayload {
  walletPubkey: string;  // player's wallet public key (string form)
  unitType: string;      // e.g. 'light-enemy', 'heavy-enemy', 'ranged-enemy'
  timestamp: number;     // Date.now() at time of kill
}

/** One entry in the on-chain leaderboard fetched via SOAR SDK */
export interface LeaderboardEntry {
  player: string;   // wallet pubkey
  score: number;    // total kill count for this session
  rank: number;     // 1-indexed rank position
}

/** Constants used by BlockchainService — store here to avoid magic strings */
export const MAGIC_ROUTER_DEVNET = 'https://devnet-router.magicblock.app/';
export const MAGIC_ROUTER_WS_DEVNET = 'wss://devnet-router.magicblock.app/';
```
