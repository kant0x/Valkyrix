<objective>
Write the Rust Anchor program `valkyrix-recorder` that records per-session gameplay stats on MagicBlock Ephemeral Rollups. The program uses `#[ephemeral]` + `#[program]` for ER compatibility. PlayerStats PDA accumulates kills, builds, units, boss outcome, and score on the ER; `undelegate` commits the final state back to Solana devnet.

Purpose: This is the real on-chain recording mechanism replacing Memo TX. TypeScript client (plan 05-03) calls it via generated IDL.
Output: `anchor/` workspace with fully implemented lib.rs ready for `anchor build`.
</objective>

    ```toml
    [lib]
    crate-type = ["cdylib", "lib"]
    name = "valkyrix_recorder"

    [features]
    no-entrypoint = []
    no-idl = []
    no-log-ix-name = []
    cpi = ["no-entrypoint"]
    default = []

    [dependencies]
    anchor-lang = { version = "0.32.1", features = ["init-if-needed"] }
    ephemeral-rollups-sdk = { version = "0.8.0", features = ["anchor", "cpi"] }
    ```

    Create `anchor/programs/valkyrix-recorder/src/lib.rs`:

    ```rust
    use anchor_lang::prelude::*;
    use ephemeral_rollups_sdk::anchor::{commit, delegate, ephemeral};
    use ephemeral_rollups_sdk::cpi::DelegateConfig;
    use ephemeral_rollups_sdk::ephem::{commit_accounts, commit_and_undelegate_accounts};

    declare_id!("PROGRAM_ID_PLACEHOLDER");

    pub const PLAYER_SEED: &[u8] = b"player";

    #[ephemeral]
    #[program]
    pub mod valkyrix_recorder {
        use super::*;

        /// Create PlayerStats PDA for a wallet. Idempotent (init-if-needed).
        pub fn initialize(ctx: Context<Initialize>) -> Result<()> {
            let stats = &mut ctx.accounts.player_stats;
            stats.wallet = ctx.accounts.payer.key();
            stats.kills = 0;
            stats.builds = 0;
            stats.units = 0;
            stats.boss_outcome = 0;
            stats.score = 0;
            stats.bump = ctx.bumps.player_stats;
            Ok(())
        }

        /// Delegate PlayerStats PDA to MagicBlock ER. Called at session start.
        pub fn delegate(ctx: Context<DelegateInput>) -> Result<()> {
            ctx.accounts.delegate_pda(
                &ctx.accounts.payer,
                &[PLAYER_SEED, ctx.accounts.payer.key().as_ref()],
                DelegateConfig::default(),
            )
        }

        /// Record an enemy kill. score += 10, kills += 1. Sent to ER during session.
        pub fn record_kill(ctx: Context<UpdateStats>, _unit_type: String) -> Result<()> {
            let stats = &mut ctx.accounts.player_stats;
            stats.kills = stats.kills.saturating_add(1);
            stats.score = stats.score.saturating_add(10);
            Ok(())
        }

        /// Record a building placed. score += 1, builds += 1. Sent to ER during session.
        pub fn record_build(ctx: Context<UpdateStats>, _building_type: String) -> Result<()> {
            let stats = &mut ctx.accounts.player_stats;
            stats.builds = stats.builds.saturating_add(1);
            stats.score = stats.score.saturating_add(1);
            Ok(())
        }

        /// Record a unit recruited. score += 1, units += 1. Sent to ER during session.
        pub fn record_unit(ctx: Context<UpdateStats>, _unit_type: String) -> Result<()> {
            let stats = &mut ctx.accounts.player_stats;
            stats.units = stats.units.saturating_add(1);
            stats.score = stats.score.saturating_add(1);
            Ok(())
        }

        /// Record boss outcome. outcome: 1=killed(+10), 2=negotiated(+10000).
        pub fn record_boss_outcome(ctx: Context<UpdateStats>, outcome: u8) -> Result<()> {
            let stats = &mut ctx.accounts.player_stats;
            stats.boss_outcome = outcome;
            let bonus: u64 = if outcome == 2 { 10_000 } else { 10 };
            stats.score = stats.score.saturating_add(bonus);
            Ok(())
        }

        /// Commit state to base chain (mid-session checkpoint, optional).
        pub fn commit(ctx: Context<CommitContext>) -> Result<()> {
            commit_accounts(
                &ctx.accounts.payer,
                vec![&ctx.accounts.player_stats.to_account_info()],
                &ctx.accounts.magic_context,
                &ctx.accounts.magic_program,
            )
        }

        /// End session: commit final state + undelegate PDA back to base chain.
        pub fn undelegate(ctx: Context<CommitContext>) -> Result<()> {
            ctx.accounts.player_stats.exit(&crate::ID)?;
            commit_and_undelegate_accounts(
                &ctx.accounts.payer,
                vec![&ctx.accounts.player_stats.to_account_info()],
                &ctx.accounts.magic_context,
                &ctx.accounts.magic_program,
            )
        }
    }

    #[derive(Accounts)]
    pub struct Initialize<'info> {
        #[account(mut)]
        pub payer: Signer<'info>,
        #[account(
            init_if_needed,
            payer = payer,
            space = 8 + PlayerStats::INIT_SPACE,
            seeds = [PLAYER_SEED, payer.key().as_ref()],
            bump,
        )]
        pub player_stats: Account<'info, PlayerStats>,
        pub system_program: Program<'info, System>,
    }

    #[delegate]
    #[derive(Accounts)]
    pub struct DelegateInput<'info> {
        pub payer: Signer<'info>,
        #[account(mut, del, seeds = [PLAYER_SEED, payer.key().as_ref()], bump)]
        pub pda: AccountInfo<'info>,
    }

    #[derive(Accounts)]
    pub struct UpdateStats<'info> {
        #[account(mut)]
        pub payer: Signer<'info>,
        #[account(mut, seeds = [PLAYER_SEED, payer.key().as_ref()], bump)]
        pub player_stats: Account<'info, PlayerStats>,
    }

    #[commit]
    #[derive(Accounts)]
    pub struct CommitContext<'info> {
        #[account(mut)]
        pub payer: Signer<'info>,
        #[account(mut, seeds = [PLAYER_SEED, payer.key().as_ref()], bump)]
        pub player_stats: Account<'info, PlayerStats>,
    }

    #[account]
    #[derive(InitSpace)]
    pub struct PlayerStats {
        pub wallet: Pubkey,   // 32
        pub kills: u32,       // 4
        pub builds: u32,      // 4
        pub units: u32,       // 4
        pub boss_outcome: u8, // 1  (0=none, 1=killed, 2=negotiated)
        pub score: u64,       // 8
        pub bump: u8,         // 1
    }
    ```

    Critical points:
    - `#[ephemeral]` MUST appear before `#[program]` — MagicBlock ER requires this ordering
    - `declare_id!("PROGRAM_ID_PLACEHOLDER")` is a placeholder; real ID assigned after `anchor build` + `anchor deploy`
    - `saturating_add` prevents integer overflow on score accumulation
    - `#[delegate]` macro on `DelegateInput` injects ER delegation accounts automatically
    - `#[commit]` macro on `CommitContext` injects `magic_context` + `magic_program` accounts automatically
    - The `del` constraint on `DelegateInput.pda` marks it as the account to delegate
    - `ctx.accounts.player_stats.exit(&crate::ID)?` in `undelegate` serialises dirty state before undelegating
