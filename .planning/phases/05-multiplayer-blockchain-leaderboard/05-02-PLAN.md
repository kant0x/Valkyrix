```typescript
import { ConnectionMagicRouter } from '@magicblock-labs/ephemeral-rollups-sdk';
import {
  Transaction,
  PublicKey,
  TransactionInstruction,
  SystemProgram,
} from '@solana/web3.js';
import { getProvider, getState } from '../wallet/WalletService';
import { MAGIC_ROUTER_DEVNET, MAGIC_ROUTER_WS_DEVNET, type KillPayload } from './blockchain.types';

// Memo program ID — standard on Solana for lightweight data anchoring
const MEMO_PROGRAM_ID = new PublicKey('MemoSq4gqABAXKb96qnH8TysNcWxMyWCqXgDLGmfcHr');

export class BlockchainService {
  private connection: ConnectionMagicRouter;

  constructor(connection?: ConnectionMagicRouter) {
    this.connection = connection ?? new ConnectionMagicRouter(MAGIC_ROUTER_DEVNET, {
      wsEndpoint: MAGIC_ROUTER_WS_DEVNET,
    });
  }

  /**
   * Fire-and-forget: DO NOT await this from the game loop.
   * Records a kill event as an on-chain TX via MagicBlock ephemeral rollup.
   * Errors are caught and logged silently — game never blocks on TX outcome.
   */
  async recordKill(unitType: string, walletPubkey: string | null): Promise<void> {
    if (!walletPubkey) return;

    const walletState = getState();
    if (!walletState.connected || !walletState.walletType) return;

    const provider = getProvider(walletState.walletType);
    if (!provider?.signTransaction) return;

    try {
      const payload: KillPayload = {
        walletPubkey,
        unitType,
        timestamp: Date.now(),
      };

      const memoData = JSON.stringify(payload);
      const feePayer = new PublicKey(walletPubkey);

      const { blockhash } = await this.connection.getLatestBlockhash();
      const tx = new Transaction({ feePayer, recentBlockhash: blockhash });

      tx.add(new TransactionInstruction({
        keys: [{ pubkey: feePayer, isSigner: true, isWritable: false }],
        programId: MEMO_PROGRAM_ID,
        data: Buffer.from(memoData, 'utf-8'),
      }));

      const signedTx = await provider.signTransaction(tx);
      // skipPreflight: true for speed; no await — fire-and-forget
      void this.connection.sendRawTransaction(
        (signedTx as unknown as { serialize(): Buffer }).serialize(),
        { skipPreflight: true }
      );
    } catch (e) {
      console.warn('[BlockchainService] kill TX failed (non-blocking):', e);
    }
  }
}
```
