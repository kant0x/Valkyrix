```typescript
import { SoarProgram } from '@magicblock-labs/soar-sdk';
import { AnchorProvider } from '@coral-xyz/anchor';
import { Connection, PublicKey } from '@solana/web3.js';
import { ConnectionMagicRouter } from '@magicblock-labs/ephemeral-rollups-sdk';
import { getProvider, getState } from '../wallet/WalletService';
import type { LeaderboardEntry } from './blockchain.types';
import { MAGIC_ROUTER_DEVNET, MAGIC_ROUTER_WS_DEVNET } from './blockchain.types';

/** Set after running scripts/init-soar-devnet.ts — null means leaderboard not initialized */
export const SOAR_LEADERBOARD_PDA: string | null = null;

export class LeaderboardService {
  private soar: SoarProgram;
  private provider: AnchorProvider;

  constructor(soarProgram?: SoarProgram, anchorProvider?: AnchorProvider) {
    const connection = new ConnectionMagicRouter(MAGIC_ROUTER_DEVNET, {
      wsEndpoint: MAGIC_ROUTER_WS_DEVNET,
    }) as unknown as Connection;

    if (soarProgram && anchorProvider) {
      this.soar = soarProgram;
      this.provider = anchorProvider;
    } else {
      this.provider = LeaderboardService.buildProvider(connection);
      this.soar = SoarProgram.getFromProvider(this.provider);
    }
  }

  private static buildProvider(connection: Connection): AnchorProvider {
    const walletState = getState();
    if (!walletState.walletType || !walletState.publicKey) {
      throw new Error('[LeaderboardService] Wallet not connected');
    }
    const raw = getProvider(walletState.walletType)!;
    const anchorWallet = {
      publicKey: new PublicKey(walletState.publicKey),
      signTransaction: (tx: unknown) => raw.signTransaction!(tx as never),
      signAllTransactions: (txs: unknown[]) => raw.signAllTransactions!(txs as never),
    };
    return new AnchorProvider(connection, anchorWallet as never, { commitment: 'confirmed' });
  }

  /**
   * Submit session aggregate kill count to SOAR leaderboard.
   * Call ONCE at session end (not per-kill — that is BlockchainService.recordKill).
   * Awaits confirmation intentionally — score submission is not latency-sensitive.
   */
  async submitScore(walletPubkey: string, killCount: number): Promise<void> {
    if (!SOAR_LEADERBOARD_PDA) {
      console.warn('[LeaderboardService] SOAR_LEADERBOARD_PDA not set — score not submitted');
      return;
    }
    try {
      const leaderboardPda = new PublicKey(SOAR_LEADERBOARD_PDA);
      const playerPubkey = new PublicKey(walletPubkey);
      const tx = await (this.soar.instructionBuilder as {
        andSubmitScoreToLeaderboard(p: PublicKey, l: PublicKey, s: number): { build(): Promise<unknown> }
      }).andSubmitScoreToLeaderboard(playerPubkey, leaderboardPda, killCount).build();
      await this.provider.sendAndConfirm(tx as never);
    } catch (e) {
      console.warn('[LeaderboardService] submitScore failed:', e);
    }
  }

  /**
   * Fetch top 10 leaderboard entries from SOAR on-chain accounts.
   * Returns [] on any error.
   */
  async fetchLeaderboard(): Promise<LeaderboardEntry[]> {
    try {
      const accounts = await (this.soar as {
        fetchAllLeaderboardAccounts(): Promise<Array<{ account: { player: { toString(): string }; score: number } }>>
      }).fetchAllLeaderboardAccounts();

      return accounts
        .sort((a, b) => b.account.score - a.account.score)
        .slice(0, 10)
        .map((a, i) => ({
          player: a.account.player.toString(),
          score: a.account.score,
          rank: i + 1,
        }));
    } catch (e) {
      console.warn('[LeaderboardService] fetchLeaderboard failed:', e);
      return [];
    }
  }
}
```

```typescript
import type { LeaderboardEntry } from '../blockchain/blockchain.types';

const STYLE_ID = 'leaderboard-overlay-styles';
const OVERLAY_ID = 'leaderboard-overlay';

const CSS = `
#leaderboard-overlay {
  position: fixed;
  top: 0; left: 0; right: 0; bottom: 0;
  background: rgba(0, 0, 0, 0.85);
  color: #e8d5a0;
  display: flex;
  flex-direction: column;
  align-items: center;
  justify-content: center;
  z-index: 200;
  font-family: 'Segoe UI', sans-serif;
}
#leaderboard-overlay h2 {
  font-size: 2rem;
  margin-bottom: 1.5rem;
  letter-spacing: 0.1em;
  text-transform: uppercase;
}
.leaderboard-row {
  display: flex;
  gap: 2rem;
  padding: 0.4rem 1rem;
  width: 400px;
  border-bottom: 1px solid rgba(232, 213, 160, 0.2);
}
.leaderboard-rank { width: 2rem; text-align: center; }
.leaderboard-player { flex: 1; font-size: 0.85rem; opacity: 0.8; }
.leaderboard-score { font-weight: bold; }
`;

export class LeaderboardOverlay {
  show(entries: LeaderboardEntry[]): void {
    this.hide(); // idempotent — remove if already showing

    if (!document.getElementById(STYLE_ID)) {
      const style = document.createElement('style');
      style.id = STYLE_ID;
      style.textContent = CSS;
      document.head.appendChild(style);
    }

    const overlay = document.createElement('div');
    overlay.id = OVERLAY_ID;
    overlay.className = 'leaderboard-overlay';

    const title = document.createElement('h2');
    title.textContent = 'Leaderboard';
    overlay.appendChild(title);

    if (entries.length === 0) {
      const empty = document.createElement('p');
      empty.textContent = 'No scores yet';
      overlay.appendChild(empty);
    } else {
      for (const entry of entries) {
        const row = document.createElement('div');
        row.className = 'leaderboard-row';

        const rankEl = document.createElement('span');
        rankEl.className = 'leaderboard-rank';
        rankEl.textContent = `#${entry.rank}`;

        const playerEl = document.createElement('span');
        playerEl.className = 'leaderboard-player';
        const truncated = entry.player.length > 8
          ? `${entry.player.slice(0, 8)}...`
          : entry.player;
        playerEl.textContent = truncated;

        const scoreEl = document.createElement('span');
        scoreEl.className = 'leaderboard-score';
        scoreEl.textContent = String(entry.score);

        row.appendChild(rankEl);
        row.appendChild(playerEl);
        row.appendChild(scoreEl);
        overlay.appendChild(row);
      }
    }

    document.body.appendChild(overlay);
  }

  hide(): void {
    const existing = document.getElementById(OVERLAY_ID);
    if (existing) existing.remove();
  }
}
```
