  <behavior>
    - initSession(walletPubkey) → calls program.methods.initialize() on baseProvider, then program.methods.delegate() on baseProvider, then waits 3000ms for ER delegation
    - initSession throws ChainUnavailableError if wallet not connected or RPC unavailable
    - recordKill(unitType) → calls erProgram.methods.recordKill(unitType) fire-and-forget, increments local kills counter
    - recordBuild(type) → calls erProgram.methods.recordBuild(type) fire-and-forget, increments local builds counter
    - recordUnit(type) → calls erProgram.methods.recordUnit(type) fire-and-forget, increments local units counter
    - recordBossOutcome(outcome: 1 | 2) → calls erProgram.methods.recordBossOutcome(outcome) fire-and-forget, stores outcome
    - All record* methods are void — `_fireAndForget()`: attempt TX; if it fails → wait 300ms → retry once → if second attempt also fails, discard silently with console.warn. The method is still void/fire-and-forget from the caller's perspective.
    - endSession() → calls erProgram.methods.undelegate(), fetches PDA account, returns { score, kills, builds, units, bossOutcome }
    - getSessionSnapshot() → returns local { kills, builds, units, bossOutcome, score } (computed: kills*10 + builds + units + bossBonus)
    - resetSessionStats() → zeroes local counters

    ```typescript
    import { AnchorProvider, Program } from '@coral-xyz/anchor';
    import { Connection, PublicKey } from '@solana/web3.js';
    import { getCurrentState, getProvider } from '../wallet/WalletService';
    import {
      MAGIC_ROUTER_DEVNET,
      MAGIC_ROUTER_WS_DEVNET,
      VALKYRIX_LEDGER_PROGRAM_ID,
    } from './blockchain.types';

    // Use generated IDL when available (after anchor build), fall back to shim
    let ValkyrixRecorderIdl: unknown;
    try {
      // eslint-disable-next-line @typescript-eslint/no-require-imports
      ValkyrixRecorderIdl = require('../../anchor/target/idl/valkyrix_recorder.json');
    } catch {
      ValkyrixRecorderIdl = (await import('./valkyrix_recorder.idl')).ValkyrixRecorderIdl;
    }

    const SOLANA_DEVNET_RPC = 'https://api.devnet.solana.com';
    const PLAYER_SEED = Buffer.from('player');
    const DELEGATE_SETTLE_MS = 3000;

    export class ChainUnavailableError extends Error {
      constructor(msg = 'Chain unavailable') {
        super(msg);
        this.name = 'ChainUnavailableError';
      }
    }

    type SessionStats = {
      kills: number;
      builds: number;
      units: number;
      bossOutcome: 0 | 1 | 2;
    };

    export class BlockchainService {
      private sessionStats: SessionStats = { kills: 0, builds: 0, units: 0, bossOutcome: 0 };
      private walletPubkey: PublicKey | null = null;
      private baseProgram: Program | null = null;
      private erProgram: Program | null = null;

      /** Called at game session start. Initializes + delegates PlayerStats PDA to ER. */
      async initSession(walletPubkeyStr: string): Promise<void> {
        const walletState = getCurrentState();
        if (!walletState.connected || !walletState.walletType) {
          throw new ChainUnavailableError('Wallet not connected');
        }
        const provider = getProvider(walletState.walletType);
        if (!provider?.signTransaction) {
          throw new ChainUnavailableError('Wallet provider unavailable');
        }

        this.walletPubkey = new PublicKey(walletPubkeyStr);

        const anchorWallet = {
          publicKey: this.walletPubkey,
          signTransaction: <T>(tx: T) => provider.signTransaction!(tx as never) as Promise<T>,
          signAllTransactions: <T>(txs: T[]) =>
            provider.signAllTransactions
              ? (provider.signAllTransactions(txs as never) as Promise<T[]>)
              : Promise.resolve(txs),
        };

        const baseProvider = new AnchorProvider(
          new Connection(SOLANA_DEVNET_RPC, 'confirmed'),
          anchorWallet as never,
          { commitment: 'confirmed' },
        );
        const erProvider = new AnchorProvider(
          new Connection(MAGIC_ROUTER_DEVNET, {
            commitment: 'confirmed',
            wsEndpoint: MAGIC_ROUTER_WS_DEVNET,
          }),
          anchorWallet as never,
          { commitment: 'confirmed' },
        );

        const programId = new PublicKey(VALKYRIX_LEDGER_PROGRAM_ID);
        this.baseProgram = new Program(ValkyrixRecorderIdl as never, baseProvider);
        this.erProgram = new Program(ValkyrixRecorderIdl as never, erProvider);

        const [playerStatsPda] = PublicKey.findProgramAddressSync(
          [PLAYER_SEED, this.walletPubkey.toBuffer()],
          programId,
        );

        // initialize (idempotent on-chain) + delegate to ER
        await (this.baseProgram.methods as never as { initialize: () => { accounts: (a: object) => { rpc: () => Promise<string> } } })
          .initialize()
          .accounts({ payer: this.walletPubkey, playerStats: playerStatsPda })
          .rpc();

        await (this.baseProgram.methods as never as { delegate: () => { accounts: (a: object) => { rpc: () => Promise<string> } } })
          .delegate()
          .accounts({ payer: this.walletPubkey, pda: playerStatsPda })
          .rpc();

        // Wait for ER delegation to propagate (MagicBlock ~1-3s)
        await new Promise<void>((resolve) => setTimeout(resolve, DELEGATE_SETTLE_MS));

        this.resetSessionStats();
      }

      /** Record an enemy kill. Fire-and-forget. score += 10 on-chain. */
      recordKill(unitType: string): void {
        this.sessionStats.kills += 1;
        this._fireAndForget('recordKill', [unitType]);
      }

      /** Record a building placed. Fire-and-forget. score += 1 on-chain. */
      recordBuild(buildingType: string): void {
        this.sessionStats.builds += 1;
        this._fireAndForget('recordBuild', [buildingType]);
      }

      /** Record a unit recruited. Fire-and-forget. score += 1 on-chain. */
      recordUnit(unitType: string): void {
        this.sessionStats.units += 1;
        this._fireAndForget('recordUnit', [unitType]);
      }

      /** Record boss outcome. outcome: 1 = killed (+10), 2 = negotiated (+10000). Fire-and-forget. */
      recordBossOutcome(outcome: 1 | 2): void {
        this.sessionStats.bossOutcome = outcome;
        this._fireAndForget('recordBossOutcome', [outcome]);
      }

      /** End session: undelegate PDA back to base chain. Returns final score from on-chain PDA. */
      async endSession(): Promise<{ score: number; kills: number; builds: number; units: number; bossOutcome: number }> {
        if (!this.erProgram || !this.walletPubkey) {
          return this._localSnapshot();
        }
        const programId = new PublicKey(VALKYRIX_LEDGER_PROGRAM_ID);
        const [playerStatsPda] = PublicKey.findProgramAddressSync(
          [PLAYER_SEED, this.walletPubkey.toBuffer()],
          programId,
        );
        try {
          await (this.erProgram.methods as never as { undelegate: () => { accounts: (a: object) => { rpc: () => Promise<string> } } })
            .undelegate()
            .accounts({ payer: this.walletPubkey, playerStats: playerStatsPda })
            .rpc();

          // Fetch final on-chain state from base chain after undelegate settles
          await new Promise<void>((resolve) => setTimeout(resolve, 2000));
          const account = await (this.baseProgram!.account as never as {
            playerStats: { fetch: (pda: PublicKey) => Promise<{ score: bigint; kills: number; builds: number; units: number; bossOutcome: number }> }
          }).playerStats.fetch(playerStatsPda);

          return {
            score: Number(account.score),
            kills: account.kills,
            builds: account.builds,
            units: account.units,
            bossOutcome: account.bossOutcome,
          };
        } catch (err) {
          console.warn('[BlockchainService] endSession failed, using local snapshot:', err);
          return this._localSnapshot();
        }
      }

      /** Local snapshot with score computed from session counters. */
      getSessionSnapshot(): { kills: number; builds: number; units: number; bossOutcome: number; score: number } {
        return this._localSnapshot();
      }

      resetSessionStats(): void {
        this.sessionStats = { kills: 0, builds: 0, units: 0, bossOutcome: 0 };
      }

      private _localSnapshot() {
        const { kills, builds, units, bossOutcome } = this.sessionStats;
        const bossBonus = bossOutcome === 2 ? 10_000 : bossOutcome === 1 ? 10 : 0;
        const score = kills * 10 + builds + units + bossBonus;
        return { kills, builds, units, bossOutcome, score };
      }

      private _fireAndForget(rpc: () => Promise<void>, instruction: string): void {
        const attempt = async () => {
          try {
            await rpc();
          } catch (err) {
            // 1 retry after 300ms, then discard silently
            await new Promise(resolve => setTimeout(resolve, 300));
            try {
              await rpc();
            } catch (retryErr) {
              console.warn(`[BlockchainService] ${instruction} TX failed after retry (non-blocking):`, retryErr);
            }
          }
        };
        void attempt();
      }
    }
    ```
