  <name>Task 2: Create SOAR one-time init script</name>
  <files>

  <action>
Create `scripts/` directory if it does not exist. Create `scripts/init-soar-devnet.ts`:

This is a one-time deployment script — NOT called at runtime. It:
1. Loads a keypair from `~/.config/solana/id.json` (standard Solana CLI keypair)
2. Connects to Solana devnet (`https://api.devnet.solana.com`)
3. Creates an `AnchorProvider` with that keypair
4. Calls `soar.initializeNewGame(...)` to register Valkyrix on the SOAR program
5. Calls `soar.addLeaderboard(...)` to create the leaderboard PDA
6. Logs the resulting `GAME_PDA` and `LEADERBOARD_PDA` addresses to stdout

```typescript
#!/usr/bin/env tsx
/**
 * One-time SOAR devnet initialization script.
 * Run: npx tsx scripts/init-soar-devnet.ts
 *
 * Requires:
 *   - Solana CLI keypair at ~/.config/solana/id.json (devnet airdrop it first)
 *   - @coral-xyz/anchor, @magicblock-labs/soar-sdk installed
 *
 * Output: GAME_PDA and LEADERBOARD_PDA — paste these into blockchain.types.ts
 */

import { Connection, Keypair, PublicKey } from '@solana/web3.js';
import { AnchorProvider, Wallet } from '@coral-xyz/anchor';
import { SoarProgram } from '@magicblock-labs/soar-sdk';
import { readFileSync } from 'fs';
import { homedir } from 'os';
import { join } from 'path';

const DEVNET_RPC = 'https://api.devnet.solana.com';

async function main(): Promise<void> {
  // Load keypair from Solana CLI default location
  const keypairPath = join(homedir(), '.config', 'solana', 'id.json');
  const keypairData = JSON.parse(readFileSync(keypairPath, 'utf-8')) as number[];
  const keypair = Keypair.fromSecretKey(Uint8Array.from(keypairData));

  console.log('Authority pubkey:', keypair.publicKey.toString());

  const connection = new Connection(DEVNET_RPC, 'confirmed');
  const provider = new AnchorProvider(connection, new Wallet(keypair), { commitment: 'confirmed' });

  const soar = SoarProgram.getFromProvider(provider);

  console.log('Initializing Valkyrix game on SOAR devnet...');

  // initializeNewGame — create the game account
  // Parameters: title, description, nftMeta (optional), leaderboardCap
  const { newGame: gamePda, transaction: gameInitTx } = await (soar as {
    initializeNewGame(title: string, desc: string, nftMeta: PublicKey | null, cap: number): Promise<{
      newGame: PublicKey;
      transaction: unknown;
    }>;
  }).initializeNewGame('Valkyrix', 'On-chain leaderboard for Valkyrix tower defense', null, 10);

  await provider.sendAndConfirm(gameInitTx as never);
  console.log('GAME_PDA:', gamePda.toString());

  // addLeaderboard — create the leaderboard PDA tied to the game
  const { newLeaderBoard: leaderboardPda, transaction: lbInitTx } = await (soar as {
    addLeaderboard(game: PublicKey, desc: string, nftMeta: PublicKey | null, scoresToRetain: number, isAscending: boolean, allowMultipleScores: boolean): Promise<{
      newLeaderBoard: PublicKey;
      transaction: unknown;
    }>;
  }).addLeaderboard(gamePda, 'Kill Leaderboard', null, 10, false, false);

  await provider.sendAndConfirm(lbInitTx as never);
  console.log('LEADERBOARD_PDA:', leaderboardPda.toString());

  console.log('\n--- COPY THESE INTO src/blockchain/blockchain.types.ts ---');
  console.log(`export const SOAR_GAME_PDA = '${gamePda.toString()}';`);
  console.log(`export const SOAR_LEADERBOARD_PDA = '${leaderboardPda.toString()}';`);
}

main().catch(console.error);
```

Note: This script uses `tsx` to run TypeScript directly. If not installed: `npm install -D tsx`. The script outputs PDAs that the developer must manually paste into `LeaderboardService.ts` to activate CHAIN-02/03.

if (!winLossShown && gameState.phase === 'won') {
    gameScreenHudRef?.showWinLossOverlay('won');
    winLossShown = true;
    // Fire-and-forget: submit score + fetch leaderboard, then show overlay

    const walletPubkey = walletState.publicKey;
    void (async () => {
