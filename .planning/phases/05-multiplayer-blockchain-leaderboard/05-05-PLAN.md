  <files>src/session/SessionLayer.ts, src/session/SessionLayer.test.ts</files>
  <behavior>
    - connect() calls getSlot() on ConnectionMagicRouter with 5000ms timeout
    - connect() resolves when getSlot() resolves within timeout
    - connect() rejects with ChainUnavailableError when getSlot() hangs past 5s
    - connect() rejects with ChainUnavailableError when getSlot() throws
    - isConnected property returns true after successful connect(), false before
    - getConnection() returns the connection after connect(), throws if not connected
  </behavior>
  <action>
    Create `src/session/SessionLayer.ts`:

    ```typescript
    import { ConnectionMagicRouter } from '@magicblock-labs/ephemeral-rollups-sdk';

    export const MAGICBLOCK_DEVNET_RPC = 'https://devnet.magicblock.app/';
    export const SOLANA_DEVNET_RPC = 'https://api.devnet.solana.com';
    const CHAIN_TIMEOUT_MS = 5000;

    type SessionConnection = Pick<ConnectionMagicRouter, 'getSlot'>;

    export class ChainUnavailableError extends Error {
      constructor() {
        super('MagicBlock devnet unavailable or timed out');
        this.name = 'ChainUnavailableError';
      }
    }

    async function createDefaultConnection(): Promise<SessionConnection> {
      return new ConnectionMagicRouter(MAGICBLOCK_DEVNET_RPC);
    }

    export class SessionLayer {
      private connection: SessionConnection | null = null;

      constructor(private readonly createConnection = createDefaultConnection) {}

      get isConnected(): boolean { return this.connection !== null; }

      async connect(): Promise<void> {
        this.connection = await this.createConnection();
        await new Promise<void>((resolve, reject) => {
          const timer = setTimeout(() => reject(new ChainUnavailableError()), CHAIN_TIMEOUT_MS);
          this.connection!.getSlot().then(() => { clearTimeout(timer); resolve(); }).catch(() => {
            clearTimeout(timer);
            this.connection = null;
            reject(new ChainUnavailableError());
          });
        });
      }

      getConnection(): SessionConnection {
        if (!this.connection) throw new Error('SessionLayer not connected — call connect() first');
        return this.connection;
      }
    }
    ```

    **main.ts — session lifecycle:**

    1. Session start (when player presses Play → game mounts):
       ```typescript
       // Instantiate once at module level:
       const blockchainService = new BlockchainService();
       let sessionScoreSubmitStarted = false;

       // On game start (async handler):
       try {
         await sessionLayer.connect();
         const { publicKey } = getCurrentState();
         if (publicKey) {
           await blockchainService.initSession(publicKey);
         }
       } catch (err) {
         if (err instanceof ChainUnavailableError) {
           showChainUnavailableScreen(); // "помехи" screen
           return;
         }
       }
       buildingSystem.setBlockchainService(blockchainService);
       combatSystem.setBlockchainService(blockchainService);
       bossSystem.setBlockchainService(blockchainService);
       recruitmentSystem.setBlockchainService(blockchainService);
       blockchainService.resetSessionStats();
       sessionScoreSubmitStarted = false;
       ```

    2. Session end (win or loss condition triggered):
       ```typescript
       async function beginSessionScoreSubmit(): Promise<void> {
         if (sessionScoreSubmitStarted) return;
         sessionScoreSubmitStarted = true;
         const { publicKey } = getCurrentState();
         if (!publicKey) return;
         // Get authoritative score from on-chain state
         let snapshot: { score: number; kills: number };
         try {
           snapshot = await blockchainService.endSession();
         } catch {
           snapshot = blockchainService.getSessionSnapshot();
         }
         void submitScoreWithRetry(publicKey, snapshot.score, snapshot.kills);
       }
       ```

    3. `submitScoreWithRetry(walletPubkey, score, kills, maxRetries = 3)`:

       - On insufficient SOL error (msg includes 'insufficient'/'lamports'/'0x1'): showScoreError('пополните баланс'); return
       - On all retries exhausted: showScoreError('Результат не сохранён в лидерборде (ошибка сети)')

    4. `showScoreError(message)`:
       - Non-blocking toast fixed at bottom-center
       - Auto-removes after 5 seconds
       - CSS: dark red background, white text

    5. Reset `sessionScoreSubmitStarted = false` when a new game starts.
