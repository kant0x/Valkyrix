<!-- Colyseus 0.17 GameRoom pattern: -->
```typescript
// server/rooms/GameRoom.ts
import { Room, Client } from 'colyseus';
import { Schema, MapSchema, type } from '@colyseus/schema';

export class GameRoomState extends Schema {
  @type({ map: 'number' })
  killCounts = new MapSchema<number>();
}

export class GameRoom extends Room<GameRoomState> {
  onCreate() {
    this.setState(new GameRoomState());

    this.onMessage('kill', (client, message: { unitType: string; walletPubkey: string }) => {
      const current = this.state.killCounts.get(message.walletPubkey) ?? 0;
      this.state.killCounts.set(message.walletPubkey, current + 1);
      // State change is automatically broadcast to all clients via Colyseus delta sync
    });
  }

  onJoin(client: Client) {
    console.log(`[GameRoom] ${client.sessionId} joined`);
  }

  onLeave(client: Client) {
    console.log(`[GameRoom] ${client.sessionId} left`);
  }
}
```

// src/session/MultiplayerClient.ts
import Colyseus from 'colyseus.js';

const WS_SERVER = 'ws://localhost:2567';

export class MultiplayerClient {
  private client = new Colyseus.Client(WS_SERVER);
  private room: Colyseus.Room | null = null;

  async connect(roomName = 'game_room'): Promise<void> {
    this.room = await this.client.joinOrCreate(roomName);
  }

  broadcastKill(unitType: string, walletPubkey: string): void {
    this.room?.send('kill', { unitType, walletPubkey });
  }

  onKillBroadcast(cb: (walletPubkey: string, killCount: number) => void): void {
    this.room?.state.killCounts.onChange((value, key) => cb(key, value));
  }

  disconnect(): void {
    this.room?.leave();
    this.room = null;
  }
}
```

import { Room, Client } from 'colyseus';
import { Schema, MapSchema, type } from '@colyseus/schema';

/** Shared room state: kill counts per wallet pubkey, broadcast to all players */
export class GameRoomState extends Schema {
  @type({ map: 'number' })
  killCounts = new MapSchema<number>();
}

export class GameRoom extends Room<GameRoomState> {
  maxClients = 4; // Valkyrix Phase 5: 2-4 players per session

  onCreate(): void {
    this.setState(new GameRoomState());

    this.onMessage<{ unitType: string; walletPubkey: string }>('kill', (client, message) => {
      if (!message?.walletPubkey) return;
      const current = this.state.killCounts.get(message.walletPubkey) ?? 0;
      this.state.killCounts.set(message.walletPubkey, current + 1);
      // Colyseus delta-syncs state changes to all clients automatically
    });
  }

  onJoin(client: Client, options: { walletPubkey?: string } = {}): void {
    console.log(`[GameRoom] ${client.sessionId} joined (wallet: ${options.walletPubkey ?? 'anon'})`);
  }

  onLeave(client: Client): void {
    console.log(`[GameRoom] ${client.sessionId} left`);
  }
}
```

Create `src/session/MultiplayerClient.ts`:
```typescript
import type Colyseus from 'colyseus.js';

const WS_SERVER = typeof process !== 'undefined' && process.env.COLYSEUS_SERVER
  ? process.env.COLYSEUS_SERVER
  : 'ws://localhost:2567';

export class MultiplayerClient {
  private client: Colyseus.Client;
  private room: Colyseus.Room | null = null;

  constructor(client?: Colyseus.Client) {
    if (client) {
      this.client = client;
    } else {
      // Dynamic import to avoid SSR/test issues — colyseus.js requires browser WebSocket
      const ColyseusModule = require('colyseus.js') as typeof import('colyseus.js');
      this.client = new ColyseusModule.Client(WS_SERVER);
    }
  }

  async connect(roomName = 'game_room'): Promise<void> {
    try {
      this.room = await this.client.joinOrCreate(roomName);
    } catch (e) {
      console.warn('[MultiplayerClient] connect failed:', e);
    }
  }

  /**
   * Send a kill event to the Colyseus room — all players see the updated kill count.
   * No-op if not connected.
   */
  broadcastKill(unitType: string, walletPubkey: string): void {
    if (!this.room) return;
    this.room.send('kill', { unitType, walletPubkey });
  }

  /**
   * Register a callback for when any player's kill count changes.
   * cb receives (walletPubkey, newCount).
   */
  onKillBroadcast(cb: (walletPubkey: string, killCount: number) => void): void {
    if (!this.room) return;
    (this.room.state as { killCounts?: { onChange?: (cb: (val: number, key: string) => void) => void } })
      .killCounts?.onChange?.((value, key) => cb(key, value));
  }

  disconnect(): void {
    this.room?.leave();
    this.room = null;
  }
}
```

import { describe, it, expect, vi, beforeEach } from 'vitest';
import { MultiplayerClient } from './MultiplayerClient';

function makeMockClient() {
  const sendMock = vi.fn();
  const leaveMock = vi.fn();
  const onChangeMock = vi.fn();

  const mockRoom = {
    send: sendMock,
    leave: leaveMock,
    state: {
      killCounts: { onChange: onChangeMock },
    },
  };

  const joinOrCreateMock = vi.fn(async () => mockRoom);
  const mockColyseusClient = { joinOrCreate: joinOrCreateMock } as never;

  return { mockColyseusClient, mockRoom, sendMock, leaveMock, onChangeMock, joinOrCreateMock };
}

describe('MultiplayerClient', () => {
  it('broadcastKill sends correct message shape when connected', async () => {
    const { mockColyseusClient, sendMock } = makeMockClient();
    const client = new MultiplayerClient(mockColyseusClient);
    await client.connect();
    client.broadcastKill('heavy-enemy', 'WalletABC');
    expect(sendMock).toHaveBeenCalledWith('kill', { unitType: 'heavy-enemy', walletPubkey: 'WalletABC' });
  });

  it('broadcastKill is a no-op when not connected (room is null)', () => {
    const { mockColyseusClient, sendMock } = makeMockClient();
    const client = new MultiplayerClient(mockColyseusClient);
    // connect() not called — room is null
    client.broadcastKill('light-enemy', 'WalletXYZ');
    expect(sendMock).not.toHaveBeenCalled();
  });

  it('onKillBroadcast registers onChange callback on killCounts', async () => {
    const { mockColyseusClient, onChangeMock } = makeMockClient();
    const client = new MultiplayerClient(mockColyseusClient);
    await client.connect();
    const cb = vi.fn();
    client.onKillBroadcast(cb);
    expect(onChangeMock).toHaveBeenCalled();
  });

  it('disconnect calls room.leave and sets room to null', async () => {
    const { mockColyseusClient, leaveMock } = makeMockClient();
    const client = new MultiplayerClient(mockColyseusClient);
    await client.connect();
    client.disconnect();
    expect(leaveMock).toHaveBeenCalled();
    // After disconnect, broadcastKill is a no-op (no send called)
    const { sendMock } = makeMockClient(); // fresh mock to verify no call
    client.broadcastKill('ranged-enemy', 'WalletDEF');
    expect(sendMock).not.toHaveBeenCalled();
  });
});
